create table user_login(
    ID int not null,
    password varchar(10),
    f_name varchar(20),
    l_name varchar(20),
    primary key (ID)
    );

create table teacher_login(
    ID int not null,
    password varchar(10),
    f_name varchar(20),
    l_name varchar(20),
    primary key (ID)
    );

create table questions(
    ID int serial not null,
    question varchar(70),
    primary key (ID)
    );

create table answers(
    ID int not null,
    correct_answer varchar(70),
	wrong_answer1 varchar(70),
	wrong_answer2 varchar(70),
	wrong_answer3 varchar(70),
    primary key (ID)
    );

create table test(
    ID int not null,
    question_id int not null,
    answer_id int not null,
    primary key (ID),
    foreign key (question_id) references questions (ID),
    foreign key (answer_id) references answers (ID)
    );

create table user_grades(
    ID int not null,
    grade int,
    test_id int not null,
    user_id int not null,
    primary key (ID),
    foreign key (test_id) references test(ID),
    foreign key (user_id) references user_login(ID)
);
   
alter table test 
add status varchar(20),
